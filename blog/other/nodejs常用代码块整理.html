# 二、文件读写

  • 文件读取：
var fs = require("fs");
// 异步读取
fs.readFile('input.txt', function (err, data) {
   if (err) {
       return console.error(err);
   }
   console.log("异步读取: " + data.toString());
});
// 同步读取
var data = fs.readFileSync('input.txt');
console.log("同步读取: " + data.toString());
console.log("程序执行完毕。");
  • 文件写入

       return console.error(err);
   }

# 三、数据库

# 1.MySql

  • 数据库连接池和公用方法:

./db_config.js

/**
 * Created by zachary on 2016/11/18.
 * MySql连接池
 */
var mysql=require("mysql");
var pool = mysql.createPool({
    host: '127.0.0.1',
    user: 'root',
    password: 'root',
    database: 'pachong',
    port: 3306
});
//带参数的查询方法
var querySQL = function(sql,vals,callback){
    pool.getConnection(function(err,conn){
        if(err){
            callback(err,null,null);
        }else{
            conn.query(sql,vals,function(qerr,vals,fields){
                //释放连接
                conn.release();
                //事件驱动回调
                callback(qerr,vals,fields);
            });
        }
    });
};
//不带参数的查询
var queryNSQL = function(sql,callback){
    pool.getConnection(function(err,conn){
        if(err){
            callback(err,null,null);
        }else{
            conn.query(sql,function(qerr,vals,fields){
                //释放连接
                conn.release();
                //事件驱动回调
                callback(qerr,vals,fields);
            });
        }
    });
};
function insertSQL(sql,val) {
    pool.getConnection(function (err, conn) {
        if(err){
            console.log(err);
        }else {
            conn.query(sql,val,function (err) {
                    console.log(err);
                });
        }
    });
}
exports.querySQL = querySQL;
exports.queryNSQL = queryNSQL;
exports.insertSQL = insertSQL;
